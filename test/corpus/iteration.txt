func main() {
    let values:int[] = [1, 2, 3]
    for value in values {
        if value == 2 {
            continue
        }
        print(value)
        break
    }
    var mutable:int[] = [1]
    for var value in mutable {
        value += 1
    }
}

    if (enabled) {}
    while enabled {}
    while (enabled) {}
    for value in values {}
    for (value in values) {}
    for (let value in values) {}
