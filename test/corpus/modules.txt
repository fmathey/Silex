use Window.Session
public use Window.Event as WindowEvent

public struct Session {}
public class Window {
  var id: int
  protected var state: int
  public var title: str
  public func show() {}
}

public func create() Window.Session {
    return Window.Session()
}

internal class Handle {
  internal var state: int
  public init() {}
  internal func join() {}
}

internal func make_handle() Handle {
  return Handle()
}
