func apply(value:int, callback:func(int) int) int {
    return callback(value)
}

func main() {
    let doubled:func(int) int = func(value:int) int {
        return value * 2
    }
    print(apply(4, doubled))
}

func use_mutable(callback:func(&int)) {
    var value = 0
    callback(&value)
}
