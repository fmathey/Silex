func inspect(state:State, pending:File?, result:Result<File,Error>) {
    match borrow state {
        loaded(file) => { print(file) }
        failed(error) => { print(error) }
    }
    match move state {
        loaded(var file) => { consume(move file) }
        failed(error) => { print(error) }
    }
    if file = borrow pending { print(file) }
    if var file = move pending { consume(move file) }
    let value = try move result
}
